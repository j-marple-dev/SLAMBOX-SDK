16/// @brief Set mavlink communication config protocol
17class SetMavlinkCommunicationConfigProtocol : public BaseProtocol {
18 public:
19 /// @brief Set mavlink communication config protocol constructor
20 /// @param enabled enabled or not
21 /// @param baudrate baudrate of mavlink communication
22 SetMavlinkCommunicationConfigProtocol(bool enabled, uint32_t baudrate)
23 : BaseProtocol(protocol::kModeSetMavlinkCommConfig),
24 enabled_(enabled),
25 baudrate_(baudrate) {
26 vector<uint8_t> baudrate_bytes = to_bytes::four(baudrate);
27 this->payload_.clear();
28 this->payload_.push_back(enabled);
29 this->payload_.insert(this->payload_.end(), baudrate_bytes.begin(),
30 baudrate_bytes.end());
31 }
32
33 /// @brief Set mavlink communication config protocol constructor
34 /// @param bytes byte array of whole protocol or payload only
35 /// @param payload_only if true, then `bytes` is not decapsulated
36 explicit SetMavlinkCommunicationConfigProtocol(const vector<uint8_t> &bytes,
37 bool payload_only = false)
38 : BaseProtocol(bytes, protocol::kModeSetMavlinkCommConfig, payload_only) {
39 this->init_from_payload_();
40 }
41
42 /// @copydoc BaseProtocol::handle_callback
43 void handle_callback(
44 const vector<ParsedMessageInterface *> &callbacks) override {
45 for (auto callback : callbacks) {
46 callback->on_set_mavlink_communication_config(this->enabled_,
47 this->baudrate_);
48 }
49 }
50
51 /// @brief Get enabled or not
52 /// @return enabled or not
53 bool get_enabled() const { return enabled_; }
54
55 /// @brief Get baudrate of mavlink communication
56 /// @return baudrate of mavlink communication
57 uint32_t get_baudrate() const { return baudrate_; }
58
59 protected:
60 /// @brief Initialize `enabled_` and `baudrate_` from `payload`
61 void init_from_payload_() override {
62 this->enabled_ = this->payload_[0];
63 this->baudrate_ = from_bytes::to_int(
64 vector<uint8_t>(this->payload_.begin() + 1, this->payload_.end()));
65 }
66
67 private:
68 /// @brief enabled or not
69 bool enabled_;
70
71 /// @brief baudrate of mavlink communication
72 uint32_t baudrate_;
73};
