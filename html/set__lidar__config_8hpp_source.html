17/// @brief Set lidar config protocol
18class SetLidarConfigProtocol : public BaseProtocol {
19 public:
20 /// @brief Set lidar config protocol constructor
21 /// @param lidar_type LiDAR type defined in `protocol::kLidarCodeToType`
22 explicit SetLidarConfigProtocol(uint8_t lidar_type)
23 : BaseProtocol(protocol::kModeSetLidarConfig), lidar_type_(lidar_type) {
24 this->payload_.clear();
25 this->payload_.push_back(lidar_type);
26 }
27
28 /// @brief Set lidar config protocol constructor
29 /// @param lidar_type LiDAR type defined in `protocol::kLidarTypeToCode`
30 explicit SetLidarConfigProtocol(std::string lidar_type)
31 : BaseProtocol(protocol::kModeSetLidarConfig) {
32 if (protocol::kLidarTypeToCode.find(lidar_type) ==
33 protocol::kLidarTypeToCode.end()) {
34 this->lidar_type_ = protocol::kLidarUnknown;
35 } else {
36 this->lidar_type_ = protocol::kLidarTypeToCode.at(lidar_type);
37 }
38
39 this->payload_.clear();
40 this->payload_.push_back(this->lidar_type_);
41 }
42
43 /// @brief Set lidar config protocol constructor
44 /// @param bytes byte array of whole protocol or payload only
45 /// @param payload_only if true, then `bytes` is not decapsulated
46 explicit SetLidarConfigProtocol(const vector<uint8_t> &bytes,
47 bool payload_only = false)
48 : BaseProtocol(bytes, protocol::kModeSetLidarConfig, payload_only) {
49 this->init_from_payload_();
50 }
51
52 /// @copydoc BaseProtocol::handle_callback
53 void handle_callback(
54 const vector<ParsedMessageInterface *> &callbacks) override {
55 for (auto callback : callbacks) {
56 callback->on_set_lidar_config(this->lidar_type_);
57 }
58 }
59
60 /// @brief Get lidar type defined in `protocol::kLidarCodeToType`
61 /// @return lidar type defined in `protocol::kLidarCodeToType`
62 uint8_t get_lidar_type() const { return lidar_type_; }
63
64 protected:
65 /// @brief Initialize `lidar_type` from `payload`
66 void init_from_payload_() override { this->lidar_type_ = this->payload_[0]; }
67
68 private:
69 /// @brief enabled or not
70 uint8_t lidar_type_;
71};
