16/// @brief Get pointcloud protocol
17class PushPointCloudProtocol : public BaseProtocol {
18 public:
19 /// @brief PushPointCloudProtocol constructor
20 /// @param pointcloud Pointcloud
21 /// @param compress_payload if true, then payload is compressed
22 explicit PushPointCloudProtocol(const flexxlam_msgs::PointCloud2 &pointcloud,
23 bool compress_payload = false)
24 : BaseProtocol(protocol::kModePushPointcloud) {
25 this->pointcloud_ = pointcloud;
26 this->payload_ =
27 flexxlam_msgs::serialize(pointcloud, /*compress=*/compress_payload);
28 }
29
30 /// @brief PushPointCloudProtocol constructor
31 /// @param bytes byte array of whole protocol (from header to checksum) or
32 /// payload
33 /// @param payload_only if true, then payload is not decapsulated
34 /// @details This constructor decapsulates byte array and then initialize
35 /// `payload` and `pointcloud`
36 explicit PushPointCloudProtocol(const vector<uint8_t> &bytes,
37 bool payload_only = false)
38 : BaseProtocol(bytes, protocol::kModePushPointcloud, payload_only) {
39 this->init_from_payload_();
40 }
41
42 /// @copydoc BaseProtocol::handle_callback
43 void handle_callback(
44 const vector<ParsedMessageInterface *> &callbacks) override {
45 for (auto callback : callbacks) {
46 callback->on_push_pointcloud(this->pointcloud_);
47 }
48 }
49
50 /// @brief Get pointcloud
51 /// @return pointcloud
52 flexxlam_msgs::PointCloud2 get_pointcloud() { return this->pointcloud_; }
53
54 protected:
55 /// @brief Initialize `pointcloud` from `payload`
56 void init_from_payload_() override {
57 if (!this->is_valid()) {
58 return;
59 }
60
61 this->pointcloud_ = flexxlam_msgs::PointCloud2(this->payload_);
62 }
63
64 private:
65 /// @brief Pointcloud
66 flexxlam_msgs::PointCloud2 pointcloud_;
67};
