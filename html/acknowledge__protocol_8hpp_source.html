18/// @brief Acknowledge protocol
19class AcknowledgeProtocol : public BaseProtocol {
20 public:
21 /// @brief Acknowledge protocol constructor
22 /// @param requested_mode requested mode to acknowledge
23 /// @param status status of requested mode
24 AcknowledgeProtocol(std::array<uint8_t, 2> requested_mode, uint8_t status)
25 : BaseProtocol(protocol::kModeAck),
26 requested_mode_(requested_mode),
27 status_(status) {
28 this->payload_.clear();
29 this->payload_.push_back(this->requested_mode_[0]);
30 this->payload_.push_back(this->requested_mode_[1]);
31 this->payload_.push_back(this->status_);
32 }
33
34 /// @brief Acknowledge protocol constructor
35 /// @param bytes byte array of whole protocol or payload only
36 /// @param payload_only if true, then `bytes` is not decapsulated
37 /// @details This constructor decapsulates byte array and then initialize
38 explicit AcknowledgeProtocol(const vector<uint8_t> &bytes,
39 bool payload_only = false)
40 : BaseProtocol(bytes, protocol::kModeAck, payload_only) {
41 this->init_from_payload_();
42 }
43
44 /// @copydoc BaseProtocol::handle_callback
45 void handle_callback(
46 const vector<ParsedMessageInterface *> &callbacks) override {
47 for (auto callback : callbacks) {
48 callback->on_acknowledge(this->requested_mode_, this->status_);
49 }
50 }
51
52 protected:
53 /// @brief Initialize `requested_mode_` and `status_` from `payload`
54 /// @details This function is called from constructor
55 void init_from_payload_() override {
56 this->requested_mode_[0] = this->payload_[0];
57 this->requested_mode_[1] = this->payload_[1];
58 this->status_ = this->payload_[2];
59 }
60
61 private:
62 /// @brief Requested mode to acknowledge
63 std::array<uint8_t, 2> requested_mode_;
64
65 /// @brief Status of requested mode
66 uint8_t status_;
67};
