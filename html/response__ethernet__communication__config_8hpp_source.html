16/// @brief Response ethernet communication config protocol
17class ResponseEthernetCommunicationConfig : public BaseProtocol {
18 public:
19 /// @brief Response ethernet communication config protocol constructor
20 /// @param enabled enabled or not
21 /// @param port port number of ethernet communication
22 ResponseEthernetCommunicationConfig(bool enabled, uint32_t port)
23 : BaseProtocol(protocol::kModeResEthernetCommConfig),
24 enabled_(enabled),
25 port_(port) {
26 vector<uint8_t> baudrate_bytes = to_bytes::four(port);
27 this->payload_.clear();
28 this->payload_.push_back(enabled);
29 this->payload_.insert(this->payload_.end(), baudrate_bytes.begin(),
30 baudrate_bytes.end());
31 }
32
33 /// @brief Response data status protocol constructor
34 /// @param bytes byte array of whole protocol or payload only
35 /// @param payload_only if true, then `bytes` is not decapsulated
36 explicit ResponseEthernetCommunicationConfig(const vector<uint8_t> &bytes,
37 bool payload_only = false)
38 : BaseProtocol(bytes, protocol::kModeResEthernetCommConfig,
39 payload_only) {
40 this->init_from_payload_();
41 }
42
43 /// @copydoc BaseProtocol::handle_callback
44 void handle_callback(
45 const vector<ParsedMessageInterface *> &callbacks) override {
46 for (auto callback : callbacks) {
47 callback->on_response_ethernet_communication_config(this->enabled_,
48 this->port_);
49 }
50 }
51
52 /// @brief Get enabled or not
53 /// @return enabled or not
54 bool get_enabled() const { return enabled_; }
55
56 /// @brief Get port number of ethernet communication
57 /// @return port number of ethernet communication
58 uint32_t get_port() const { return port_; }
59
60 protected:
61 /// @brief Initialize `enabled_` and `baudrate_` from `payload`
62 void init_from_payload_() override {
63 this->enabled_ = this->payload_[0];
64 this->port_ = from_bytes::to_int(
65 vector<uint8_t>(this->payload_.begin() + 1, this->payload_.end()));
66 }
67
68 private:
69 /// @brief enabled or not
70 bool enabled_;
71
72 /// @brief baudrate of ethernet communication
73 uint32_t port_;
74};
