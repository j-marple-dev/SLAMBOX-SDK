14vector<uint8_t> from_string(std::string str) {
15 int size = str.size();
16 if (str.size() > 255) {
17 LOG(WARNING) << "String size is too big. Only first 255 characters will be "
18 "sent.";
19 size = 255;
20 }
21 vector<uint8_t> bytes;
22 bytes.push_back(static_cast<uint8_t>(size));
23 bytes.insert(bytes.end(), str.begin(), str.begin() + size);
24
25 return bytes;
26}

60vector<uint8_t> four(uint32_t x) {

63 bytes.push_back((x >> 16) & 0xFF);

89uint32_t to_int(vector<uint8_t> bytes) {
90 uint32_t x = 0;
91
92 int size = bytes.size();
93
94 if (size > 4) {
95 LOG(WARNING) << "The size of the byte array is larger than 4. "
96 "Only the first 4 bytes are used.";
97 size = 4;
98 }
99
100 for (int i = 0; i < size; i++) {
101 x += bytes[i] << (8 * (size - i - 1));
102 }
103
104 return x;
105}

107std::string to_string(vector<uint8_t> bytes, size_t offset) {
108 if (bytes.size() < offset + 1) {
109 LOG(WARNING) << "The size of the byte array is smaller than the offset. "
110 "Empty string is returned.";
111 return "";
112 }
113
114 int size = bytes[offset];
115
116 if ((bytes.size() - offset) < size + 1) {
117 LOG(WARNING) << "The size of the byte array is smaller than the string "
118 "size. Only the first "
119 << bytes.size() - 1 << " bytes are used. The string size is "
120 << size << ".";
121 size = bytes.size() - offset - 1;
122 }
123
124 std::string str(bytes.begin() + offset + 1,
125 bytes.begin() + offset + 1 + size);
126
127 return str;
128}
