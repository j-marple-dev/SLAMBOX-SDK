15vector<uint8_t> BaseProtocol::encapsulate_(vector<uint8_t> payload) {
16 int length = payload.size();
17
18 std::vector<uint8_t> length_bytes = to_bytes::two(length);
19 std::array<uint8_t, 2> checksum = BaseProtocol::get_checksum(
20 std::vector<uint8_t>(this->mode_.begin(), this->mode_.end()),
21 length_bytes, payload);
22
23 vector<uint8_t> bytes(length + 8, 0);
24
25 bytes[0] = protocol::kHeader[0];
26 bytes[1] = protocol::kHeader[1];
27
28 bytes[2] = this->mode_[0];
29 bytes[3] = this->mode_[1];
30
31 bytes[4] = length_bytes[0];
32 bytes[5] = length_bytes[1];
33
34 memcpy(bytes.data() + 6, payload.data(), length);
35
36 bytes[length + 6] = checksum[0];
37 bytes[length + 7] = checksum[1];
38
39 return bytes;
40}

42vector<uint8_t> BaseProtocol::decapsulate_(vector<uint8_t> bytes) {
43 uint8_t header[2] = {bytes[0], bytes[1]};
44 if (memcmp(header, protocol::kHeader.begin(), 2) != 0) {
45 this->is_valid_ = false;
46 return vector<uint8_t>();
47 }
48
49 vector<uint8_t> mode = {bytes[2], bytes[3]};
50 vector<uint8_t> length = {bytes[4], bytes[5]};
51 int len = length[0] * 256 + length[1];
52
53 vector<uint8_t> payload(len, 0);
54 memcpy(payload.data(), bytes.data() + 6, len);
55
56 std::array<uint8_t, 2> checksum_in_msg = {bytes[6 + len], bytes[7 + len]};
57 std::array<uint8_t, 2> checksum =
58 BaseProtocol::get_checksum(mode, length, payload);
59
60 if (memcmp(checksum_in_msg.data(), checksum.data(), 2) != 0) {
61 this->is_valid_ = false;
62 return vector<uint8_t>();
63 }
64
65 return payload;
66}
