32void FlexXlamParser::parse(vector<uint8_t> data) {
33 // 1. Add data to buffer
34 this->buffer_.insert(this->buffer_.end(), data.begin(), data.end());
35
36 // 2. Remove data if buffer size is larger than buffer size
37 if (this->buffer_.size() > this->buffer_size_) {
38 this->buffer_.erase(
39 this->buffer_.begin(),
40 this->buffer_.begin() + this->buffer_.size() - this->buffer_size_);
41 }
42
43 while (this->buffer_.size() > 7) {
44 // 3. Find header
45 if (!this->find_header_()) {
46 return;
47 }
48
49 // 4. Check length
50 std::array<uint8_t, 2> mode = {this->buffer_[2], this->buffer_[3]};
51 vector<uint8_t> length_bytes(this->buffer_.begin() + 4,
52 this->buffer_.begin() + 6);
53 uint32_t length = from_bytes::to_int(length_bytes);
54
55 if (!this->is_length_valid_({mode[0], mode[1]}, length)) {
56 this->buffer_.erase(this->buffer_.begin(), this->buffer_.begin() + 2);
57 continue;
58 }
59
60 if (this->buffer_.size() < length + 8) {
61 return;
62 }
63
64 // 5. Check checksum
65 vector<uint8_t> payload(this->buffer_.begin() + 6,
66 this->buffer_.begin() + 6 + length);
67 std::array<uint8_t, 2> computed_checksum =
68 BaseProtocol::get_checksum({mode[0], mode[1]}, length_bytes, payload);
69 std::array<uint8_t, 2> checksum = {this->buffer_[6 + length],
70 this->buffer_[7 + length]};
71
72 // if (computed_checksum != checksum) {
73 if (memcmp(computed_checksum.data(), checksum.data(), 2) != 0) {
74 this->buffer_.erase(this->buffer_.begin(), this->buffer_.begin() + 2);
75 continue;
76 }
77
78 // 6. Decapsulate payload
79 this->decapsulate_payload_(mode, payload);
80
81 // 7. Remove data from buffer
82 this->buffer_.erase(this->buffer_.begin(),
83 this->buffer_.begin() + 6 + length);
84 }
85}

87bool FlexXlamParser::find_header_() {
88 bool is_header_found = false;
89 while (this->buffer_.size() > 7) {
90 std::vector<uint8_t> header_candidate(this->buffer_.begin(),
91 this->buffer_.begin() + 2);
92 if (memcmp(header_candidate.data(), protocol::kHeader.data(), 2) == 0) {
93 is_header_found = true;
94 break;
95 }
96 this->buffer_.erase(this->buffer_.begin());
97 }
98
99 return is_header_found;
100}

161bool FlexXlamParser::is_length_valid_(std::array<uint8_t, 2> mode,
162 uint16_t length) {
163 if (protocol::kModeToPayloadLengthEq.find(mode) !=
164 protocol::kModeToPayloadLengthEq.end()) {
165 return length == protocol::kModeToPayloadLengthEq.at(mode);
166 } else if (protocol::kModeToPayloadLengthGt.find(mode) !=
167 protocol::kModeToPayloadLengthGt.end()) {
168 return length > protocol::kModeToPayloadLengthGt.at(mode);
169 }
170
171 return false;
172}
