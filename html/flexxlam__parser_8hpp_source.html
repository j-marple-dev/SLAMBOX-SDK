113/// @brief Protocol parser for FlexXlam Driver
114class FlexXlamParser {
115 public:
116 /// @brief Protocol parser constructor
117 /// @param buffer_size size of buffer
118 explicit FlexXlamParser(size_t buffer_size = 1024)
119 : buffer_size_(buffer_size) {
120 buffer_.reserve(buffer_size_ * 2);
121 }
122
123 /// @brief Parse byte array
124 /// @param data byte array
125 /// @details This function parses byte array and then call
126 /// parsed_message_callback.
127 void parse(vector<uint8_t> data);
128
129 /// @brief Add parsed message callback
130 /// @param parsed_message_interface parsed message interface
131 void add_parsed_message_callback(
132 ParsedMessageInterface *parsed_message_interface);
133
134 private:
135 /// @brief Find header in `buffer_`
136 /// @return true if header is found
137 /// @details This function finds header in `buffer_` while erasing data
138 /// before header.
139 bool find_header_();
140
141 /// @brief Decapsulate payload and call parsed message callback
142 /// @param mode mode
143 /// @param payload payload
144 /// @details This function decapsulates payload and then call registered
145 /// callbacks.
146 void decapsulate_payload_(const std::array<uint8_t, 2> &mode,
147 const vector<uint8_t> &payload);
148
149 /// @brief Check if length is valid by comparing with mode
150 /// @param mode mode
151 /// @param length length
152 /// @return true if length is valid
153 bool is_length_valid_(std::array<uint8_t, 2> mode, uint16_t length);
154
155 /// @brief Byte array for decapsulation
156 vector<uint8_t> buffer_;
157
158 /// @brief Size of buffer
159 size_t buffer_size_;
160
161 /// @brief Parsed message interface callbacks
162 vector<ParsedMessageInterface *> parsed_message_interfaces_;
163};
