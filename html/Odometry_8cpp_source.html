47std::vector<uint8_t> serialize(const Odometry &odom) {

49 msgs_to_bytes[0] = odom.pose.position.x;

52 msgs_to_bytes[3] = odom.pose.orientation.x;

54 msgs_to_bytes[5] = odom.pose.orientation.z;
55 msgs_to_bytes[6] = odom.pose.orientation.w;
56 std::memcpy(msgs_to_bytes.data() + 7, odom.pose.covariance.cbegin(),
57 36 * sizeof(float));
58 msgs_to_bytes[43] = odom.twist.linear.x;
59 msgs_to_bytes[44] = odom.twist.linear.y;
60 msgs_to_bytes[45] = odom.twist.linear.z;
61 msgs_to_bytes[46] = odom.twist.angular.x;
62 msgs_to_bytes[47] = odom.twist.angular.y;
63 msgs_to_bytes[48] = odom.twist.angular.z;
64 std::memcpy(msgs_to_bytes.data() + 49, odom.twist.covariance.cbegin(),
65 36 * sizeof(float));
66
67 std::vector<uint8_t> bytes = four(msgs_to_bytes);
68
69 vector<uint8_t> secs = four(odom.timestamp_sec);
70 vector<uint8_t> nsecs = four(odom.timestamp_nsec);
71
72 bytes.insert(bytes.begin(), nsecs.begin(), nsecs.end());
73 bytes.insert(bytes.begin(), secs.begin(), secs.end());
74
75 return bytes;
