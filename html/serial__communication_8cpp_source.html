13bool SerialCommunication::open_() {
14 LibSerial::BaudRate baud_rate_libserial = LibSerial::BaudRate::BAUD_115200;
15
16 if (SerialCommunication::kBaudRateMap.find(this->baud_rate_) !=
17 SerialCommunication::kBaudRateMap.end()) {
18 baud_rate_libserial =
19 SerialCommunication::kBaudRateMap.at(this->baud_rate_);
20 } else {
21 LOG(ERROR) << "Invalid baud rate: " << this->baud_rate_;
22 return false;
23 }
24
25 try {
26 if (this->serial_port_.IsOpen() &&
27 this->serial_port_.GetBaudRate() == baud_rate_libserial) {
28 LOG(INFO) << "Serial port is already opened";
29 return true;
30 }
31
32 if (!this->serial_port_.IsOpen()) {
33 this->serial_port_.Open(this->port_name_);
34 }
35
36 this->serial_port_.SetBaudRate(baud_rate_libserial);
37 this->serial_port_.SetCharacterSize(LibSerial::CharacterSize::CHAR_SIZE_8);
38 this->serial_port_.SetFlowControl(
39 LibSerial::FlowControl::FLOW_CONTROL_NONE);
40 this->serial_port_.SetParity(LibSerial::Parity::PARITY_NONE);
41 this->serial_port_.SetStopBits(LibSerial::StopBits::STOP_BITS_1);
42 } catch (const LibSerial::OpenFailed &) {
43 LOG(ERROR) << "Serial port " << this->port_name_ << " with baudrate "
44 << this->baud_rate_ << " failed to open";
45 return false;
46 }
47
48 return true;
49}

52bool SerialCommunication::set_baudrate(int baud_rate) {
53 if (SerialCommunication::kBaudRateMap.find(baud_rate) ==
54 SerialCommunication::kBaudRateMap.end()) {
55 LOG(ERROR) << "Invalid baud rate: " << baud_rate;
56 return false;
57 }
58
59 if (!this->serial_port_.IsOpen()) {
60 LOG(ERROR) << "Serial port is not opened";
61 return false;
62 }
63
64 try {
65 this->serial_port_.SetBaudRate(
66 SerialCommunication::kBaudRateMap.at(baud_rate));
67 this->baud_rate_ = baud_rate;
68 } catch (std::exception &e) {
69 LOG(ERROR) << "Setting baudrate failed " << baud_rate << " with exception "
70 << e.what();
71 return false;
72 }
73
74 return true;
75}

87void SerialCommunication::run() {
88 if (!this->open_()) {
89 LOG(ERROR) << "Serial port " << this->port_name_ << " with baudrate "
90 << this->baud_rate_ << " failed to open";
91 return;
92 }
93
94 this->is_running_ = true;
95 this->run_thread_ = std::thread(&SerialCommunication::run_, this);
96}

98void SerialCommunication::stop() {
99 this->is_running_ = false;

101 this->run_thread_.join();

103

110void SerialCommunication::run_() {
111 while (this->is_running_) {
112 std::vector<uint8_t> data;
113
114 try {
115 this->serial_port_.Read(data, 0, 2);
116 } catch (const LibSerial::ReadTimeout &) {
117 }
118
119 if (data.size() == 0) {
120 std::this_thread::sleep_for(std::chrono::milliseconds(1));
121 continue;
122 }
123
124 if (this->callback_ != nullptr) {
125 this->callback_(data);
126 }
127 }
128}
