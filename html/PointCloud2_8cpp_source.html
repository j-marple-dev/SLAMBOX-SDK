26PointCloud2::PointCloud2(const std::vector<uint8_t> &bytes) {
27 std::shared_ptr<vector<uint8_t>> bytes_ptr;
28 bool compression_flag = bytes[0] == 0x01;
29 uint32_t idx = 1;
30
31 if (compression_flag) {
32 uint32_t original_length =
33 to_int(vector<uint8_t>(bytes.begin() + 1, bytes.begin() + 5));
34 bytes_ptr = std::make_shared<vector<uint8_t>>(
35 decompress_bytes(vector<uint8_t>(bytes.begin() + 5, bytes.end()),
36 /*original_size=*/original_length));
37 idx = 0;
38 } else {
39 bytes_ptr = std::make_shared<vector<uint8_t>>(bytes);
40 }
41
42 const vector<uint8_t> &bytes_ = *bytes_ptr;
43
44 this->timestamp_sec =
45 to_int(vector<uint8_t>(bytes_.begin() + idx, bytes_.begin() + idx + 4));
46 this->timestamp_nsec = to_int(
47 vector<uint8_t>(bytes_.begin() + idx + 4, bytes_.begin() + idx + 8));
48 idx += 8;
49
50 this->frame_id = to_string(bytes_, idx);
51 idx += this->frame_id.length() + 1;
52
53 this->height =
54 to_int(vector<uint8_t>(bytes_.begin() + idx, bytes_.begin() + idx + 4));
55 this->width = to_int(
56 vector<uint8_t>(bytes_.begin() + idx + 4, bytes_.begin() + idx + 8));
57 idx += 8;
58
59 uint8_t n_fields = bytes_[idx++];
60 this->fields.resize(n_fields);
61 for (uint8_t i = 0; i < n_fields; i++) {
62 this->fields[i].name = to_string(bytes_, idx);
63 idx += this->fields[i].name.length() + 1;
64 this->fields[i].offset =
65 to_int(vector<uint8_t>(bytes_.begin() + idx, bytes_.begin() + idx + 4));
66 idx += 4;
67 this->fields[i].datatype = bytes_[idx++];
68 this->fields[i].count =
69 to_int(vector<uint8_t>(bytes_.begin() + idx, bytes_.begin() + idx + 4));
70 idx += 4;
71 }
72 this->is_bigendian = bytes_[idx++];
73 this->point_step =
74 to_int(vector<uint8_t>(bytes_.begin() + idx, bytes_.begin() + idx + 4));
75 this->row_step = to_int(
76 vector<uint8_t>(bytes_.begin() + idx + 4, bytes_.begin() + idx + 8));
77 idx += 8;
78
79 uint32_t n_data =
80 to_int(vector<uint8_t>(bytes_.begin() + idx, bytes_.begin() + idx + 4));
81 idx += 4;
82 this->data =
83 vector<uint8_t>(bytes_.begin() + idx, bytes_.begin() + idx + n_data);
84 idx += n_data;
85 this->is_dense = bytes_[idx++];
86}

88std::vector<uint8_t> serialize(const PointCloud2 &pointcloud, bool compress) {
89 vector<uint8_t> bytes;
90 bytes.reserve(pointcloud.data.size() + 128);
91
92 vector<uint8_t> secs = four(pointcloud.timestamp_sec);
93 vector<uint8_t> nsecs = four(pointcloud.timestamp_nsec);
94 vector<uint8_t> frame_id = from_string(pointcloud.frame_id);
95
96 vector<uint8_t> height = four(pointcloud.height);
97 vector<uint8_t> width = four(pointcloud.width);
98 uint8_t n_fields = pointcloud.fields.size();
99
100 bytes.insert(bytes.end(), secs.begin(), secs.end());
101 bytes.insert(bytes.end(), nsecs.begin(), nsecs.end());
102 bytes.insert(bytes.end(), frame_id.begin(), frame_id.end());
103 bytes.insert(bytes.end(), height.begin(), height.end());
104 bytes.insert(bytes.end(), width.begin(), width.end());
105 bytes.insert(bytes.end(), n_fields);
106 for (uint8_t i = 0; i < n_fields; i++) {
107 vector<uint8_t> name = from_string(pointcloud.fields[i].name);
108 vector<uint8_t> offset = four(pointcloud.fields[i].offset);
109 uint8_t datatype = pointcloud.fields[i].datatype;
110 vector<uint8_t> count = four(pointcloud.fields[i].count);
111
112 bytes.insert(bytes.end(), name.begin(), name.end());
113 bytes.insert(bytes.end(), offset.begin(), offset.end());
114 bytes.insert(bytes.end(), datatype);
115 bytes.insert(bytes.end(), count.begin(), count.end());
116 }
117 vector<uint8_t> point_step = four(pointcloud.point_step);
118 vector<uint8_t> row_step = four(pointcloud.row_step);
119 bytes.insert(bytes.end(), pointcloud.is_bigendian);
120 bytes.insert(bytes.end(), point_step.begin(), point_step.end());
121 bytes.insert(bytes.end(), row_step.begin(), row_step.end());
122
123 uint32_t n_data_ = pointcloud.data.size();
124 vector<uint8_t> n_data = four(n_data_);
125 bytes.insert(bytes.end(), n_data.begin(), n_data.end());
126 bytes.insert(bytes.end(), pointcloud.data.begin(), pointcloud.data.end());
127 bytes.insert(bytes.end(), pointcloud.is_dense);
128
129 if (compress) {
130 vector<uint8_t> compressed_bytes = compress_bytes(bytes);
131 vector<uint8_t> original_length = four(static_cast<uint32_t>(bytes.size()));
132 compressed_bytes.insert(compressed_bytes.begin(), original_length.begin(),
133 original_length.end());
134 compressed_bytes.insert(compressed_bytes.begin(), 0x01);
135
136 return compressed_bytes;
137 } else {
138 bytes.insert(bytes.begin(), 0x00);
139
140 return bytes;
141 }
142}
