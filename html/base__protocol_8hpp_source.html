111/// @brief Base protocol for communication
112class BaseProtocol {
113 public:
114 /// @brief Base protocol constructor
115 /// @param mode mode of protocol (2 bytes)
116 /// @details This constructor initialize mode and payload length.
117 explicit BaseProtocol(const std::array<uint8_t, 2> mode) : mode_(mode) {
118 this->is_valid_ = true;
119 }
120
121 /// @brief Base protocol constructor
122 /// @param bytes byte array
123 /// @param mode mode of protocol (2 bytes)
124 /// @param payload_only if true, then payload is not decapsulated
125 /// @details This constructor decapsulates byte array and then initialize
126 /// payload.
127 explicit BaseProtocol(const vector<uint8_t> &bytes,
128 std::array<uint8_t, 2> mode, bool payload_only = false)
129 : BaseProtocol(mode) {
130 if (payload_only) {
131 this->payload_ = bytes;
132 } else {
133 this->payload_ = decapsulate_(bytes);
134 }
135 }
136
137 /// @brief Base protocol destructor
138 /// @details This destructor deletes payload.
139 ~BaseProtocol() {}
140
141 /// @brief Get payload only
142 /// @return payload
143 vector<uint8_t> get_payload() { return this->payload_; }
144
145 /// @brief Get the validity of the protocol.
146 /// @return true if the protocol is valid, otherwise false
147 bool is_valid() { return this->is_valid_; }
148
149 /// @brief Encapsulate payload into the protocol.
150 /// @return byte array of whole protocol (from header to checksum)
151 /// @details The protocol is as follows:
152 ///
153 /// | header | mode | length | payload | checksum |
154 /// |--------|--------|--------|---------|----------|
155 /// | 2 byte | 2 byte | 2 byte | length | 1 byte |
156 virtual vector<uint8_t> encapsulate() { return encapsulate_(this->payload_); }
157
158 /// @brief Call callback functions from parsed message interface
159 /// @param callbacks parsed message interfaces
160 /// @details This function is called from `FlexXlamParser::parse`.
161 virtual void handle_callback(
162 const vector<ParsedMessageInterface *> &callbacks) = 0;
163
164 /// @brief Get checksum of protocol
165 /// @param mode mode of protocol (2 bytes)
166 /// @param length length of payload
167 /// @param payload payload
168 /// @return checksum
169 /// @details The checksum is calculated as follows:
170 /// checksum = mode[0] ^ mode[1] ^ length[0] ^ length[1] ^ payload[0] ^
171 /// payload[1] ^ ... ^ payload[length - 1]
172 static std::array<uint8_t, 2> get_checksum(vector<uint8_t> mode,
173 vector<uint8_t> length,
174 vector<uint8_t> payload) {
175 uint8_t checksum_a = mode[0];
176 uint8_t checksum_b = checksum_a;
177
178 checksum_a += mode[1];
179 checksum_b += checksum_a;
180
181 checksum_a += length[0];
182 checksum_b += checksum_a;
183
184 checksum_a += length[1];
185 checksum_b += checksum_a;
186
187 int len = length[0] * 256 + length[1];
188
189 for (int i = 0; i < len; i++) {
190 checksum_a += payload[i];
191 checksum_b += checksum_a;
192 }
193
194 return {checksum_a, checksum_b};
195 }
196
197 protected:
198 /// @brief mode of protocol (2 bytes)
199 std::array<uint8_t, 2> mode_;
200
201 /// @brief is valid protocol
202 bool is_valid_ = false;
203
204 /// @brief payload
205 // vector<uint8_t> payload = nullptr;
206 vector<uint8_t> payload_;
207
208 /// @brief Create whole protocol message from payload
209 /// @param payload payload
210 /// @return byte array
211 vector<uint8_t> encapsulate_(vector<uint8_t> payload);
212
213 /// @brief Decapsulate payload from protocol
214 /// @param bytes byte array
215 /// @return payload if checksum and header are valid, otherwise nullptr
216 vector<uint8_t> decapsulate_(vector<uint8_t> bytes);
217
218 /// @brief Initialize members of the derived class from payload
219 virtual void init_from_payload_() = 0;
220};
