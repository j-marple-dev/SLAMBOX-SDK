18/// @brief Save PCD protocol
19class CommandSavePCDProtocol : public BaseProtocol {
20 public:
21 /// @brief Command save PCD protocoll constructor
22 /// @param save save pcd or not
23 /// @param reset reset the pointcloud from buffer or not
24 explicit CommandSavePCDProtocol(bool save, bool reset)
25 : BaseProtocol(protocol::kModeCmdSavePCD), save_(save), reset_(reset) {
26 this->payload_.push_back(save_);
27 this->payload_.push_back(reset_);
28 }
29
30 /// @brief Command save PCD protocoll constructor
31 /// @param bytes byte array of whole protocol or payload
32 /// @param payload_only if trure, then bytes parameter contains only payload
33 explicit CommandSavePCDProtocol(const std::vector<uint8_t> &bytes,
34 bool payload_only = false)
35 : BaseProtocol(bytes, protocol::kModeCmdSavePCD, payload_only) {
36 this->init_from_payload_();
37 }
38
39 /// @copydoc BaseProtocol::handle_callback
40 void handle_callback(
41 const vector<ParsedMessageInterface *> &callbacks) override {
42 for (auto callback : callbacks) {
43 callback->on_command_save_pcd(this->save_, this->reset_);
44 }
45 }
46
47 /// @brief Get save(save or not)
48 /// @return save flag
49 bool get_save() { return this->save_; }
50
51 /// @brief Get reset(reset or not)
52 /// @return reset flag
53 bool get_reset() { return this->reset_; }
54
55 protected:
56 /// @brief initialize save_ and reset_ from payload
57 void init_from_payload_() override {
58 if (!this->is_valid()) {
59 return;
60 }
61 this->save_ = (this->payload_[0] == 1);
62 this->reset_ = (this->payload_[1] == 1);
63 }
64
65 private:
66 /// @brief a flag used for saving the pcd map data
67 bool save_;
68 /// @brief a flag used for reset the pointcloud buffer of odometry algorithm
69 bool reset_;
70};
